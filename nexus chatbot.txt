<script>
  // ─── STATE ───────────────────────────────────────
  let messages = [];
  let totalTokens = 0;
  let isStreaming = false;

  // ─── INIT ─────────────────────────────────────────
  window.onload = () => {
    const saved = localStorage.getItem('nexus_apikey');
    if (saved) document.getElementById('apiKeyInput').value = saved;

    const ta = document.getElementById('userInput');
    ta.addEventListener('keydown', e => {
      if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); }
    });
    ta.addEventListener('input', () => {
      ta.style.height = 'auto';
      ta.style.height = Math.min(ta.scrollHeight, 160) + 'px';
      const words = ta.value.trim().split(/\s+/).filter(Boolean).length;
      document.getElementById('liveTokens').textContent = Math.round(words * 1.3);
    });
  };

  // ─── MODAL ───────────────────────────────────────
  function openModal() { document.getElementById('modalOverlay').classList.add('active'); }
  function closeModal() { document.getElementById('modalOverlay').classList.remove('active'); }
  function saveKey() {
    const key = document.getElementById('apiKeyInput').value.trim();
    if (key) { localStorage.setItem('nexus_apikey', key); closeModal(); showToast('API key saved!'); }
  }
  document.getElementById('modalOverlay').addEventListener('click', e => { if (e.target === e.currentTarget) closeModal(); });

  // ─── SEND MESSAGE ─────────────────────────────────
  async function sendMessage() {
    if (isStreaming) return;
    const apiKey = localStorage.getItem('nexus_apikey');
    if (!apiKey) { openModal(); return; }

    const input = document.getElementById('userInput');
    const text = input.value.trim();
    if (!text) return;

    input.value = '';
    input.style.height = 'auto';
    document.getElementById('liveTokens').textContent = '0';

    // Hide welcome
    const welcome = document.getElementById('welcomeScreen');
    if (welcome) welcome.remove();

    // Add user message
    messages.push({ role: 'user', content: text });
    appendMessage('user', text);
    updateStats();

    // Show typing
    const typingEl = showTyping();
    isStreaming = true;
    toggleSendBtn(false);

    try {
      const model = document.getElementById('modelSelect').value;
      const temp = parseFloat(document.getElementById('tempSlider').value);
      const sysPrompt = document.getElementById('systemPrompt').value.trim();

      const payload = {
        model,
        temperature: temp,
        stream: true,
        messages: [
          ...(sysPrompt ? [{ role: 'system', content: sysPrompt }] : []),
          ...messages
        ]
      };

      const res = await fetch('https://openrouter.ai/api/v1/chat/completions', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json',
          'HTTP-Referer': window.location.href,
          'X-Title': 'NEXUS AI Terminal'
        },
        body: JSON.stringify(payload)
      });

      if (!res.ok) {
        const err = await res.json();
        throw new Error(err.error?.message || `HTTP ${res.status}`);
      }

      typingEl.remove();
      const aiEl = appendMessage('ai', '');
      const bubbleContent = aiEl.querySelector('.bubble-text');

      const reader = res.body.getReader();
      const decoder = new TextDecoder();
      let fullText = '';

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        const chunk = decoder.decode(value);
        const lines = chunk.split('\n').filter(l => l.startsWith('data: '));

        for (const line of lines) {
          const data = line.slice(6).trim();
          if (data === '[DONE]') continue;
          try {
            const parsed = JSON.parse(data);
            const delta = parsed.choices?.[0]?.delta?.content || '';
            if (delta) {
              fullText += delta;
              bubbleContent.innerHTML = formatMessage(fullText);
              scrollToBottom();
            }
            const usage = parsed.usage;
            if (usage) { totalTokens += usage.total_tokens || 0; updateStats(); }
          } catch (_) {}
        }
      }

      messages.push({ role: 'assistant', content: fullText });
      updateStats();

    } catch (err) {
      typingEl?.remove();
      appendMessage('ai', `⚠ Error: ${err.message}`, true);
    }

    isStreaming = false;
    toggleSendBtn(true);
  }

  // ─── RENDER ───────────────────────────────────────
  function appendMessage(role, text, isError = false) {
    const container = document.getElementById('messages');
    const time = new Date().toLocaleTimeString('en-US', { hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit' });

    const row = document.createElement('div');
    row.className = `msg-row ${role}`;

    const avatarChar = role === 'ai' ? '⬡' : '◈';
    const metaName = role === 'ai' ? 'NEXUS' : 'YOU';

    row.innerHTML = `
      <div class="avatar ${role}">${avatarChar}</div>
      <div class="bubble-wrap">
        <div class="msg-meta">${metaName} · ${time}</div>
        <div class="bubble ${isError ? 'style="border-color:var(--accent3)"' : ''}">
          <span class="bubble-text">${formatMessage(text)}</span>
        </div>
      </div>
    `;

    container.appendChild(row);
    scrollToBottom();
    return row;
  }

  function showTyping() {
    const container = document.getElementById('messages');
    const wrap = document.createElement('div');
    wrap.className = 'msg-row ai';
    wrap.innerHTML = `
      <div class="avatar ai">⬡</div>
      <div class="bubble-wrap">
        <div class="msg-meta">NEXUS · processing...</div>
        <div class="typing-bubble">
          <div class="typing-dot"></div>
          <div class="typing-dot"></div>
          <div class="typing-dot"></div>
        </div>
      </div>
    `;
    container.appendChild(wrap);
    scrollToBottom();
    return wrap;
  }

  function formatMessage(text) {
    // Code blocks
    text = text.replace(/```(\w*)\n?([\s\S]*?)```/g, (_, lang, code) =>
      `<pre><code>${escHtml(code.trim())}</code></pre>`
    );
    // Inline code
    text = text.replace(/`([^`]+)`/g, '<code>$1</code>');
    // Bold
    text = text.replace(/\*\*(.+?)\*\*/g, '<strong style="color:var(--accent2)">$1</strong>');
    // Italic
    text = text.replace(/\*(.+?)\*/g, '<em>$1</em>');
    // Newlines
    text = text.replace(/\n/g, '<br>');
    return text;
  }

  function escHtml(s) {
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  }

  // ─── UTILS ────────────────────────────────────────
  function scrollToBottom() {
    const m = document.getElementById('messages');
    m.scrollTop = m.scrollHeight;
  }

  function toggleSendBtn(enabled) {
    const btn = document.getElementById('sendBtn');
    btn.style.opacity = enabled ? '1' : '0.4';
    btn.style.cursor = enabled ? 'pointer' : 'not-allowed';
  }

  function updateStats() {
    document.getElementById('msgCount').textContent = messages.length;
    document.getElementById('tokenCount').textContent = totalTokens > 999
      ? (totalTokens / 1000).toFixed(1) + 'k'

  }

    const input = document.getElementById('userInput');

  function exportChat() {
    if (!messages.length) return;
    const content = messages.map(m => `[${m.role.toUpperCase()}]\n${m.content}`).join('\n\n---\n\n');
    const blob = new Blob([content], { type: 'text/plain' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = `nexus-chat-${Date.now()}.txt`;
    a.click();
  }

  function copyLast() {
    const aiMsgs = messages.filter(m => m.role === 'assistant');
    if (aiMsgs.length === 0) return;
    navigator.clipboard.writeText(aiMsgs[aiMsgs.length - 1].content);
    showToast('Copied to clipboard!');
  }

  function showToast(msg) {
    const t = document.createElement('div');
    t.textContent = msg;
    Object.assign(t.style, {
      position:'fixed', bottom:'30px', right:'30px', zIndex:'9999',
      background:'var(--accent)', color:'var(--bg)',
      fontFamily:"'Share Tech Mono',monospace", fontSize:'12px',
      padding:'10px 18px', borderRadius:'6px',
      boxShadow:'var(--glow)', letterSpacing:'1px',
      animation:'fadeUp 0.3s ease-out'
    });
    document.body.appendChild(t);
    setTimeout(() => t.remove(), 2500);
  }
